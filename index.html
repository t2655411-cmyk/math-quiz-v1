<script>

    // 1. 这里是你的例题植入区（你可以随时在此增加或修改题目）

    const myExampleQuestions = [

        { qText: "5 + 3 =", ans: 8 },

        { qText: "12 - 4 =", ans: 8 },

        { qText: "9 + ( ) = 15", ans: 6 },

        { qText: "2 + 3 + 4 =", ans: 9 },

        { qText: "20 - 10 =", ans: 10 },

        { qText: "( ) - 5 = 10", ans: 15 }

    ];

 

    let questions = [...myExampleQuestions];

 

    // 2. 渲染题目

    function initQuiz() {

        const container = document.getElementById('quiz-form');

        questions.forEach((q, i) => {

            const div = document.createElement('div');

            div.className = 'question-item';

            div.innerHTML = `

                <span class="question-text">${q.qText}</span>

                <input type="number" class="answer-input" id="input-${i}">

                <div class="result-icon" id="icon-${i}"></div>

            `;

            container.appendChild(div);

        });

    }

 

    // 3. 判分逻辑

    function calculateScore() {

        let score = 0;

        questions.forEach((q, index) => {

            const inputField = document.getElementById(`input-${index}`);

            const iconField = document.getElementById(`icon-${index}`);

            const userVal = parseInt(inputField.value);

 

            // 清除之前的状态

            inputField.classList.remove('correct', 'wrong');

 

            if (userVal === q.ans) {

                score++;

                inputField.classList.add('correct');

                iconField.innerText = '✅';

            } else {

                inputField.classList.add('wrong');

                iconField.innerText = '❌';

            }

        });

 

        // 计算并显示分数

        const scoreBox = document.getElementById('score-display');

        scoreBox.style.display = 'block';

        const finalScore = Math.round((score / questions.length) * 100);

        

        let message = "";

        if(finalScore === 100) message = "�� 太棒了，你是数学小天才！";

        else if(finalScore >= 60) message = "�� 做得不错，继续加油！";

        else message = "�� 没关系，再检查一下错题吧。";

 

        scoreBox.innerHTML = `<strong>得分：${finalScore}</strong><br><small>${message}</small>`;

        

        // 自动滚动到底部

        window.scrollTo({ top: document.body.scrollHeight, behavior: 'smooth' });

    }

 

    // 启动

    initQuiz();

</script>
